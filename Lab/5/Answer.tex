\section*{Problem 1}
\subsection*{a.}

\begin{align*}
		(\lambda f . \lambda g . f(g\ 1) (\lambda x . x + 4)) (\lambda y . 3 - y)
		&= (\lambda g . (g\ 1) + 4) (\lambda y . 3 - y) \\
		&= (3 - 1) + 4 \\
		&= 6
\end{align*}

\subsection*{b.}
\begin{align*}
		(\lambda f . \lambda g . f(g\ 1) (\lambda x . x + 4)) (\lambda y . 3 - y)
		&= \lambda f . f(3 - 1) (\lambda x . x + 4) \\
		&= (3 - 1) + 4 \\
		&= 6
\end{align*}

\section*{Problem 5}
For example, $a * b * c$ can be evaluated in following two ways:
$$(a * b) * c$$
$$a * (b * c)$$
Therefore the grammar is ambiguous.
Unambiguous grammar:
\begin{align*}
	<S> &::=\ <X> \\
	<X> &::=\ <X> * <Y>\ |\ <Y> \\
	<Y> &::= a\ b\ c
\end{align*}
